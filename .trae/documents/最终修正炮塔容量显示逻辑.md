    public updateCapacityView() {
        if (!this.CapacityAm) return;
        const gridData = this.getGroupedGridData();
        const children = this.CapacityAm.children;
        
        let dataIndex = 0;
        // 倒着遍历 children (从索引大的到底部开始)
        for (let i = children.length - 1; i >= 0; i--) {
            const child = children[i];
            // 即将发射的 (dataIndex=0) 放在最下面 (i=last)
            if (dataIndex < gridData.length) {
                child.active = true;
                const gridComp = child.getComponent(gridCmpt);
                if (gridComp) {
                    gridComp.setType(gridData[dataIndex].type);
                    gridComp.setCount(gridData[dataIndex].count);
                }
                dataIndex++;
            } else {
                // 数据用完后，隐藏前面的（顶部的）节点
                child.active = false;
            }
        }
    }
